TouchCap swipe
    Attributes
        ID   : 5
        Scope: local
        Value: 0
    
    Events
        Touch Press Event
            swipestore.en=1 // Start swipestore timer
            dim=brightness.val // Set brightness to saved value (from HA)
            dimtimer.en=1 // Enable timer to calc down for dimming screen
            sleeptimer.en=1
        
        Touch Release Event
            swipestore.en=0
            // Touch has ended, x
            if(tch0==0)
            {
                swipec=swipex-tch2
                // From Left to Right
                if(swipec>swipedx)
                {
                    page buttonpage04
                }
                // Right to Left
                swipec2=0-swipedx
                if(swipec<swipec2)
                {
                    page buttonpage01
                }
            }
            // Touch has ended, y
            if(tch1==0)
            {
                swipec=swipey-tch3
                // From Up to Down
                if(swipec>100)
                {
                    page buttonpage02
                }
                // Down to Up
                swipec2=0-swipedy
                if(swipec<-100)
                {
                    page buttonpage03
                }
            }
